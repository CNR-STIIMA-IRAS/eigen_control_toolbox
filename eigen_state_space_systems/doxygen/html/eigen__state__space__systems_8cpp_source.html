68 {
69  /*
70  *
71  * y(0) = C*x(0) + D u(0)
72  * y(1) = C*A*x(0) + D u(1) + C*B*u(0)
73  * y(2) = C*A^2*x(0) + D u(2) + C*B*u(1)+C*A*B*u(1)
74  * ......
75  * y(n) = C*A^(n-1)*x(0) + D u(n) + sum(j=0:n-1) C*A^j*B*u(n-1-j)
76  *
77  * Y=[y(0) ... y(n)]
78  * U=[u(0) ... u(n)]
79  * Y=obsv(A,C)*x(0)+ioMatrix(A,B,C,D)*U
80  * x(0)= obsv(A,C) \ ( Y-ioMatrix(A,B,C,D)*U )
81  *
82  * x(1) = A*x(0)+B*u(0)
83  * ......
84  * x(n) = A*x(n-1)+B*u(n-1)
85  */
86 
87  if (past_inputs.rows() != m_nin*m_order)
88  throw std::invalid_argument(("[ DiscreteStateSpace ] past_inputs has wrong dimenstion: "+std::to_string(past_inputs.rows())+" instead of "+std::to_string(m_nin*m_order)).c_str());
89 
90  if (past_outputs.rows() != m_nout*m_order)
91  throw std::invalid_argument("[ DiscreteStateSpace ] past_outputs has wrong dimenstion");
92 
93  Eigen::MatrixXd obsv=computeObservatibilityMatrix(m_A,m_C);
94  Eigen::MatrixXd io_matrix=computeInputToOutputMatrix(m_A,m_B,m_C,m_D);
